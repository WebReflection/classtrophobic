Zero Runtime, Babel Proof, Classes.

const Class = require('classtrophobic');

const List = Class({
  extends: Array,
  static: {import:(o) => List.from(o)},
  constructor(...args) {
    this.super();
    this.push(...args);
  },
  push(...args) {
    this.super.push(...args);
    return this;
  },
  get size() {
    return this.length;
  }
});
